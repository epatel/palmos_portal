    backup_time: int = 0,

    num_records: int = 0,
) -> bytes:

    return struct.pack(
        ">32sHHIIIIIIII4s4sIIH",
        name_bytes,
        attributes,
        version,
        creation_time,
        modification_time,
        backup_time,
        modification_number,
        app_info_offset,
        sort_info_offset,
        unique_id_seed,  # unique_id_seed is at offset 68 but pack order follows struct
        next_record_list,
        num_records,
        db_type.encode("ascii"),
        creator.encode("ascii"),
        unique_id_seed,
        next_record_list,
        num_records,
    )


def make_pdb_bytes(
    name: str = "TestDB",
    db_type: str = "DATA",
    creator: str = "test",
    records: list[bytes] | None = None,
    app_info: bytes | None = None,
) -> bytes:
    """Build a complete PDB file with optional records and app info."""
    records = records or []
    num_records = len(records)

    # Calculate offsets
    header_size = 78  # fixed header
    record_list_size = num_records * 8  # 8 bytes per record entry
    # 2 bytes padding after record list if there are records
    padding = 2 if num_records > 0 else 0

    app_info_offset = 0
    data_start = header_size + record_list_size + padding
    if app_info is not None:
        app_info_offset = data_start
        data_start += len(app_info)

    # Build record list entries
    record_list = b""
    offset = data_start
    for i, rec in enumerate(records):
        # offset (4 bytes) + attributes (1 byte) + unique_id (3 bytes)
        unique_id = i + 1
        record_list += struct.pack(">IB", offset, 0x40)  # 0x40 = dirty flag
        record_list += struct.pack(">I", unique_id)[1:]  # 3-byte unique ID
        offset += len(rec)

    # Build header
    name_bytes = name.encode("ascii")[:31].ljust(32, b"\x00")
    now = PALM_EPOCH_OFFSET + 1000000  # arbitrary timestamp
    header = struct.pack(
        ">32sHHIIIIII4s4sIIH",
        name_bytes,
        0x0000,  # attributes (record DB, not resource)
        1,  # version
        now,  # creation time
        now,  # modification time
        0,  # backup time
        0,  # modification number
        app_info_offset,
        0,  # sort_info_offset
        db_type.encode("ascii"),
        creator.encode("ascii"),
        0,  # unique_id_seed
        0,  # next_record_list
        num_records,
    )

    # Assemble file
    data = header + record_list
    if padding:
        data += b"\x00" * padding
    if app_info is not None:
        data += app_info
    for rec in records:
        data += rec

    return data

```python
"""PalmOS database (.pdb/.prc) file format parser and serializer."""

from __future__ import annotations

import struct
from dataclasses import dataclass, field
from datetime import datetime, timezone, timedelta

# PalmOS epoch is 1904-01-01 00:00:00 UTC (same as classic Mac OS)
_PALM_EPOCH = datetime(1904, 1, 1, tzinfo=timezone.utc)
_UNIX_EPOCH = datetime(1970, 1, 1, tzinfo=timezone.utc)
_PALM_EPOCH_OFFSET = int((_UNIX_EPOCH - _PALM_EPOCH).total_seconds())

# Database attribute flags
ATTR_RESOURCE = 0x0001  # Resource DB (.prc) vs Record DB (.pdb)
ATTR_READ_ONLY = 0x0002
ATTR_APPINFO_DIRTY = 0x0004
ATTR_BACKUP = 0x0008
ATTR_COPY_PREVENTION = 0x0040

# Header: 32s name, H attributes, H version, I crdate, I moddate,
#         I backupdate, I modnum, I appinfo_off, I sortinfo_off,
#         4s type, 4s creator, I uniqueidseed, I nextreclist, H numrecs
_HEADER_FORMAT = ">32sHHIIIIII4s4sIIH"
_HEADER_SIZE = struct.calcsize(_HEADER_FORMAT)  # 78 bytes

# Record entry: I offset, B attributes, 3-byte unique ID
_RECORD_ENTRY_SIZE = 8

# Resource entry: 4s type, H id, I offset
_RESOURCE_ENTRY_FORMAT = ">4sHI"
_RESOURCE_ENTRY_SIZE = 10


def _palm_to_datetime(palm_time: int) -> datetime:
    """Convert PalmOS timestamp to Python datetime."""
    if palm_time == 0:
        return _PALM_EPOCH
    unix_time = palm_time - _PALM_EPOCH_OFFSET
    return datetime.fromtimestamp(unix_time, tz=timezone.utc)


def _datetime_to_palm(dt: datetime) -> int:
    """Convert Python datetime to PalmOS timestamp."""
    unix_time = int(dt.timestamp())
    return unix_time + _PALM_EPOCH_OFFSET


@dataclass
class Record:
    """A record in a PDB (record database)."""
    data: bytes
    attributes: int = 0
    unique_id: int = 0


@dataclass
class Resource:
    """A resource in a PRC (resource database)."""
    res_type: str  # 4-char type code
    res_id: int
    data: bytes


@dataclass
class PalmDatabase:
    """A PalmOS database — either record-based (.pdb) or resource-based (.prc)."""
    name: str = ""
    attributes: int = 0
    version: int = 0
    creation_time: datetime = field(default_factory=lambda: datetime.now(timezone.utc))
    modification_time: datetime = field(default_factory=lambda: datetime.now(timezone.utc))
    backup_time: datetime = field(default_factory=lambda: _PALM_EPOCH)
    modification_number: int = 0
    db_type: str = "DATA"
    creator: str = "    "
    unique_id_seed: int = 0
    app_info: bytes | None = None
    sort_info: bytes | None = None
    records: list[Record] = field(default_factory=list)
    resources: list[Resource] = field(default_factory=list)

    @property
    def is_resource_db(self) -> bool:
        return bool(self.attributes & ATTR_RESOURCE)

    @classmethod
    def from_bytes(cls, data: bytes) -> PalmDatabase:
        """Parse a PDB/PRC file from raw bytes."""
        if len(data) < _HEADER_SIZE:
            raise ValueError(f"File too small: {len(data)} bytes (need at least {_HEADER_SIZE})")

        (
            name_raw, attributes, version,
            crdate, moddate, backupdate, modnum,
            appinfo_off, sortinfo_off,
            type_raw, creator_raw,
            unique_id_seed, next_rec_list, num_records,
        ) = struct.unpack_from(_HEADER_FORMAT, data, 0)

        db = cls(
            name=name_raw.split(b"\x00", 1)[0].decode("ascii", errors="replace"),
            attributes=attributes,
            version=version,
            creation_time=_palm_to_datetime(crdate),
            modification_time=_palm_to_datetime(moddate),
            backup_time=_palm_to_datetime(backupdate),
            modification_number=modnum,
            db_type=type_raw.decode("ascii", errors="replace"),
            creator=creator_raw.decode("ascii", errors="replace"),
            unique_id_seed=unique_id_seed,
        )

        is_resource = bool(attributes & ATTR_RESOURCE)
        entry_size = _RESOURCE_ENTRY_SIZE if is_resource else _RECORD_ENTRY_SIZE
        list_start = _HEADER_SIZE

        # Parse record/resource list to get offsets
        entries: list[tuple] = []
        for i in range(num_records):
            entry_offset = list_start + i * entry_size
            if is_resource:
                res_type_raw, res_id, data_offset = struct.unpack_from(
                    _RESOURCE_ENTRY_FORMAT, data, entry_offset
                )
                entries.append((data_offset, res_type_raw.decode("ascii", errors="replace"), res_id))
            else:
                data_offset = struct.unpack_from(">I", data, entry_offset)[0]
                attr = data[entry_offset + 4]
                uid_bytes = data[entry_offset + 5 : entry_offset + 8]
                unique_id = (uid_bytes[0] << 16) | (uid_bytes[1] << 8) | uid_bytes[2]
                entries.append((data_offset, attr, unique_id))

        # Read app_info block
        if appinfo_off > 0:
            # AppInfo ends at sort_info, first record/resource, or EOF
            end_candidates = [len(data)]
            if sortinfo_off > 0:
                end_candidates.append(sortinfo_off)
            if entries:
                end_candidates.append(entries[0][0])
            app_info_end = min(e for e in end_candidates if e > appinfo_off)
            db.app_info = data[appinfo_off:app_info_end]

        # Read sort_info block
        if sortinfo_off > 0:
            end_candidates = [len(data)]
            if entries:
                end_candidates.append(entries[0][0])
            sort_info_end = min(e for e in end_candidates if e > sortinfo_off)
            db.sort_info = data[sortinfo_off:sort_info_end]

        # Read record/resource data
        for i, entry in enumerate(entries):
            offset = entry[0]
            # Data extends to next entry's offset or EOF
            if i + 1 < len(entries):
                end = entries[i + 1][0]
            else:
                end = len(data)
            chunk = data[offset:end]

            if is_resource:
                _, res_type, res_id = entry
                db.resources.append(Resource(res_type=res_type, res_id=res_id, data=chunk))
            else:
                _, attr, unique_id = entry
                db.records.append(Record(data=chunk, attributes=attr, unique_id=unique_id))

        return db

    def to_bytes(self) -> bytes:
        """Serialize to PDB/PRC file format."""
        is_resource = self.is_resource_db
        items = self.resources if is_resource else self.records
        num_items = len(items)
        entry_size = _RESOURCE_ENTRY_SIZE if is_resource else _RECORD_ENTRY_SIZE

        # Calculate offsets
        list_end = _HEADER_SIZE + num_items * entry_size
        # Padding: 2 bytes after record list (standard Palm convention)
        padding = 2 if num_items > 0 else 0
        data_start = list_end + padding

        app_info_offset = 0
        sort_info_offset = 0
        current = data_start

        if self.app_info is not None:
            app_info_offset = current
            current += len(self.app_info)
        if self.sort_info is not None:
            sort_info_offset = current
            current += len(self.sort_info)

        # Calculate per-item offsets
        item_offsets: list[int] = []
        for item in items:
            item_offsets.append(current)
            current += len(item.data)

        # Build header
        name_bytes = self.name.encode("ascii")[:31].ljust(32, b"\x00")
        header = struct.pack(
            _HEADER_FORMAT,
            name_bytes,
            self.attributes,
            self.version,
            _datetime_to_palm(self.creation_time),
            _datetime_to_palm(self.modification_time),
            _datetime_to_palm(self.backup_time),
            self.modification_number,
            app_info_offset,
            sort_info_offset,
            self.db_type.encode("ascii")[:4].ljust(4),
            self.creator.encode("ascii")[:4].ljust(4),
            self.unique_id_seed,
            0,  # next_record_list
            num_items,
        )

        # Build record/resource list
        entry_list = b""
        for i, item in enumerate(items):
            if is_resource:
                res = item  # type: Resource
                entry_list += struct.pack(
                    _RESOURCE_ENTRY_FORMAT,
                    res.res_type.encode("ascii")[:4].ljust(4),
                    res.res_id,
                    item_offsets[i],
                )
            else:
                rec = item  # type: Record
                entry_list += struct.pack(">IB", item_offsets[i], rec.attributes)
                uid = rec.unique_id
                entry_list += bytes([(uid >> 16) & 0xFF, (uid >> 8) & 0xFF, uid & 0xFF])

        # Assemble
        result = header + entry_list
        if padding:
            result += b"\x00" * padding
        if self.app_info is not None:
            result += self.app_info
        if self.sort_info is not None:
            result += self.sort_info
        for item in items:
            result += item.data

        return result

    @classmethod
    def from_file(cls, path: str) -> PalmDatabase:
        """Load from a .pdb/.prc file on disk."""
        with open(path, "rb") as f:
            return cls.from_bytes(f.read())

    def to_file(self, path: str) -> None:
        """Save to a .pdb/.prc file on disk."""
        with open(path, "wb") as f:
            f.write(self.to_bytes())
```

def make_prc_bytes(
    name: str = "TestApp",
    db_type: str = "appl",
    creator: str = "test",
    resources: list[tuple[str, int, bytes]] | None = None,
) -> bytes:
    """Build a complete PRC file. resources: list of (type, id, data)."""
    resources = resources or []
    num_resources = len(resources)

    header_size = 78
    resource_list_size = num_resources * 10  # 10 bytes per resource entry
    padding = 2 if num_resources > 0 else 0
    data_start = header_size + resource_list_size + padding

    # Build resource list entries
    resource_list = b""
    offset = data_start
    for res_type, res_id, res_data in resources:
        resource_list += struct.pack(
            ">4sHI",
            res_type.encode("ascii"),
            res_id,
            offset,
        )
        offset += len(res_data)

    # Build header — attributes has ATTR_RESOURCE (0x0001) set
    name_bytes = name.encode("ascii")[:31].ljust(32, b"\x00")

    header = struct.pack(

        1,  # version
        now, now, 0, 0, 0, 0,
        db_type.encode("ascii"),
        creator.encode("ascii"),
        0, 0,
        num_resources,
    )

    data = header + resource_list
    if padding:
        data += b"\x00" * padding
    for _, _, res_data in resources:
        data += res_data

    return data

```python
"""Serial Link Protocol (SLP) — packet framing for PalmOS HotSync."""

from __future__ import annotations

import struct
from dataclasses import dataclass

# SLP signature bytes
SLP_SIGNATURE = b"\xbe\xef\xed"

# Packet types
SLP_TYPE_DATA = 0x00
SLP_TYPE_LOOPBACK = 0x01
SLP_TYPE_ACK = 0x02

# DLP socket ID
SLP_SOCKET_DLP = 3

_HEADER_SIZE = 10  # 3 sig + 1 dest + 1 src + 1 type + 2 len + 1 txn + 1 cksum


def crc16(data: bytes) -> int:
    """CRC-CCITT (polynomial 0x1021, init 0x0000)."""
    crc = 0x0000
    for byte in data:
        crc ^= byte << 8
        for _ in range(8):
            if crc & 0x8000:
                crc = (crc << 1) ^ 0x1021
            else:
                crc <<= 1
            crc &= 0xFFFF
    return crc


@dataclass
class SLPPacket:
    """A parsed SLP packet."""
    dest: int
    src: int
    ptype: int
    txn_id: int
    data: bytes

    @classmethod
    def from_bytes(cls, raw: bytes) -> SLPPacket:
        """Parse an SLP packet from raw bytes. Validates signature, checksum, and CRC."""
        if len(raw) < _HEADER_SIZE + 2:  # header + CRC
            raise ValueError(f"Packet too short: {len(raw)} bytes")

        if raw[:3] != SLP_SIGNATURE:
            raise ValueError(f"Invalid SLP signature: {raw[:3].hex()}")

        dest = raw[3]
        src = raw[4]
        ptype = raw[5]
        data_len = struct.unpack(">H", raw[6:8])[0]
        txn_id = raw[8]
        header_cksum = raw[9]

        # Verify header checksum
        expected_cksum = sum(raw[:9]) % 256
        if header_cksum != expected_cksum:
            raise ValueError(
                f"Header checksum mismatch: got 0x{header_cksum:02X}, "
                f"expected 0x{expected_cksum:02X}"
            )

        # Extract body
        body = raw[_HEADER_SIZE : _HEADER_SIZE + data_len]
        if len(body) != data_len:
            raise ValueError(f"Body truncated: got {len(body)}, expected {data_len}")

        # Verify CRC
        crc_offset = _HEADER_SIZE + data_len
        received_crc = struct.unpack(">H", raw[crc_offset : crc_offset + 2])[0]
        computed_crc = crc16(body)
        if received_crc != computed_crc:
            raise ValueError(
                f"CRC mismatch: got 0x{received_crc:04X}, expected 0x{computed_crc:04X}"
            )

        return cls(dest=dest, src=src, ptype=ptype, txn_id=txn_id, data=body)


class SLPSocket:
    """Send and receive SLP packets over a Connection."""

    def __init__(self, connection):
        self.connection = connection

    @staticmethod
    def build_packet(dest: int, src: int, ptype: int, txn_id: int, data: bytes) -> bytes:
        """Build a raw SLP packet ready for transmission."""
        data_len = len(data)
        header = struct.pack(
            ">3sBBBHB",
            SLP_SIGNATURE,
            dest, src, ptype,
            data_len,
            txn_id,
        )
        cksum = sum(header) % 256
        header += bytes([cksum])

        crc = crc16(data)
        crc_bytes = struct.pack(">H", crc)

        return header + data + crc_bytes

    def send(self, dest: int, src: int, ptype: int, txn_id: int, data: bytes) -> None:
        """Build and send an SLP packet."""
        packet = self.build_packet(dest, src, ptype, txn_id, data)
        self.connection.write(packet)

    def receive(self) -> SLPPacket:
        """Read and parse an SLP packet from the connection.

        Scans the byte stream for the 0xBEEFED signature, then reads
        the complete packet.
        """
        # Scan for signature
        buf = b""
        while True:
            byte = self.connection.read(1)
            if not byte:
                raise TimeoutError("Timeout waiting for SLP packet")
            buf += byte
            if buf.endswith(SLP_SIGNATURE):
                break
            # Keep only last 2 bytes to check for partial signature match
            if len(buf) > 3:
                buf = buf[-2:]

        # Read rest of header (7 bytes: dest, src, type, len(2), txn, cksum)
        rest_of_header = self.connection.read(7)
        if len(rest_of_header) < 7:
            raise TimeoutError("Timeout reading SLP header")

        header = SLP_SIGNATURE + rest_of_header
        data_len = struct.unpack(">H", rest_of_header[3:5])[0]

        # Read body + CRC
        body_and_crc = self.connection.read(data_len + 2)
        if len(body_and_crc) < data_len + 2:
            raise TimeoutError("Timeout reading SLP body")

        raw = header + body_and_crc
        return SLPPacket.from_bytes(raw)
```

```python
"""Packet Assembly/Disassembly Protocol (PADP) — reliable delivery for PalmOS HotSync."""

from __future__ import annotations

import struct
import logging

from palm.slp import SLPSocket, SLPPacket, SLP_TYPE_DATA, SLP_TYPE_ACK, SLP_SOCKET_DLP

logger = logging.getLogger(__name__)

# PADP packet types
PADP_TYPE_DATA = 0x01
PADP_TYPE_ACK = 0x02
PADP_TYPE_TICKLE = 0x04

# PADP fragment flags
PADP_FLAG_FIRST = 0x80
PADP_FLAG_LAST = 0x40
PADP_FLAG_MORE = 0x20  # set on non-last fragments (alternative to checking LAST)

# Maximum PADP payload per SLP packet
PADP_MAX_PAYLOAD = 1024

# Retry settings
PADP_MAX_RETRIES = 3
PADP_RETRY_TIMEOUT = 2.0


def parse_padp_header(data: bytes) -> tuple[int, int, int]:
    """Parse a PADP header. Returns (type, flags, payload_size)."""
    if len(data) < 4:
        raise ValueError(f"PADP header too short: {len(data)} bytes")
    ptype, flags, size = struct.unpack(">BBH", data[:4])
    return ptype, flags, size


def fragment_payload(data: bytes, max_size: int = PADP_MAX_PAYLOAD) -> list[tuple[int, bytes]]:
    """Split data into PADP fragments. Returns list of (flags, chunk)."""
    if not data:
        return [(PADP_FLAG_FIRST | PADP_FLAG_LAST, b"")]

    fragments = []
    offset = 0
    total = len(data)

    while offset < total:
        chunk = data[offset : offset + max_size]
        flags = 0
        if offset == 0:
            flags |= PADP_FLAG_FIRST
        if offset + len(chunk) >= total:
            flags |= PADP_FLAG_LAST
        fragments.append((flags, chunk))
        offset += len(chunk)

    return fragments


def reassemble_fragments(payloads: list[bytes]) -> bytes:
    """Concatenate fragment payloads back into the original data."""
    return b"".join(payloads)


class PADPConnection:
    """Reliable message delivery over SLP with fragmentation and ACK."""

    def __init__(self, slp: SLPSocket):
        self.slp = slp
        self._txn_id = 0xFF  # Start at 0xFF, wraps to 0x01

    def _next_txn_id(self) -> int:
        self._txn_id += 1
        if self._txn_id > 0xFF:
            self._txn_id = 0x01  # Skip 0x00
        return self._txn_id

    @staticmethod
    def build_padp_header(ptype: int, flags: int, payload_size: int) -> bytes:
        """Build a 4-byte PADP header."""
        return struct.pack(">BBH", ptype, flags, payload_size)

    def send(self, data: bytes) -> None:
        """Send data reliably, fragmenting if needed. Waits for ACK per fragment."""
        fragments = fragment_payload(data)
        txn_id = self._next_txn_id()

        for flags, chunk in fragments:
            padp_header = self.build_padp_header(PADP_TYPE_DATA, flags, len(chunk))
            slp_body = padp_header + chunk

            for attempt in range(PADP_MAX_RETRIES):
                self.slp.send(
                    dest=SLP_SOCKET_DLP, src=SLP_SOCKET_DLP,
                    ptype=SLP_TYPE_DATA, txn_id=txn_id, data=slp_body,
                )

                # Wait for ACK
                try:
                    ack_pkt = self.slp.receive()
                    if ack_pkt.ptype == SLP_TYPE_DATA:
                        ack_type, _, _ = parse_padp_header(ack_pkt.data)
                        if ack_type == PADP_TYPE_ACK:
                            break
                        if ack_type == PADP_TYPE_TICKLE:
                            logger.debug("Got tickle, retrying...")
                            continue
                    elif ack_pkt.ptype == SLP_TYPE_ACK:
                        break
                except TimeoutError:
                    logger.debug(f"Timeout waiting for ACK (attempt {attempt + 1})")
            else:
                raise TimeoutError(f"No ACK after {PADP_MAX_RETRIES} retries")

    def receive(self) -> bytes:
        """Receive a complete message, reassembling fragments. Sends ACKs."""
        payloads: list[bytes] = []

        while True:
            pkt = self.slp.receive()

            if pkt.ptype != SLP_TYPE_DATA:
                continue

            ptype, flags, size = parse_padp_header(pkt.data)

            if ptype == PADP_TYPE_TICKLE:
                # Respond to tickle with a tickle ACK
                ack_header = self.build_padp_header(PADP_TYPE_ACK, 0, 0)
                self.slp.send(
                    dest=SLP_SOCKET_DLP, src=SLP_SOCKET_DLP,
                    ptype=SLP_TYPE_DATA, txn_id=pkt.txn_id, data=ack_header,
                )
                continue

            if ptype == PADP_TYPE_DATA:
                payload = pkt.data[4 : 4 + size]
                payloads.append(payload)

                # Send ACK
                ack_header = self.build_padp_header(PADP_TYPE_ACK, flags, size)
                self.slp.send(
                    dest=SLP_SOCKET_DLP, src=SLP_SOCKET_DLP,
                    ptype=SLP_TYPE_DATA, txn_id=pkt.txn_id, data=ack_header,
                )

                if flags & PADP_FLAG_LAST:
                    break

        return reassemble_fragments(payloads)
```

from __future__ import annotations

import struct
import logging
import array

import usb.core
import usb.util

logger = logging.getLogger(__name__)

# Handspring Visor USB IDs
VISOR_VENDOR_ID = 0x082D
VISOR_PRODUCT_ID = 0x0100

# CMP (Connection Management Protocol) types
CMP_TYPE_INIT = 0x01
CMP_TYPE_ABORT = 0x02
CMP_TYPE_EXTENDED = 0x03

# CMP flags
CMP_FLAG_CHANGE_BAUD = 0x80

# CMP packet: type(1) + flags(1) + ver_major(1) + ver_minor(1) + unused(2) + max_baud(4)
_CMP_FORMAT = ">BBBBHI"
_CMP_SIZE = struct.calcsize(_CMP_FORMAT)


class Connection:
    """USB bulk transfer connection to a Handspring Visor."""

    VENDOR_ID = VISOR_VENDOR_ID
    PRODUCT_ID = VISOR_PRODUCT_ID

    def __init__(self):
        self._dev: usb.core.Device | None = None
        self._ep_in = None
        self._ep_out = None

    def open(self) -> None:
        """Find the Visor, claim the interface, discover endpoints, and do CMP handshake."""
        self._dev = usb.core.find(idVendor=self.VENDOR_ID, idProduct=self.PRODUCT_ID)
        if self._dev is None:
            raise ConnectionError(
                "No Handspring Visor found. "
                "Is the device in the cradle? Press HotSync and try again."
            )

        # Detach kernel driver if needed
        try:
            if self._dev.is_kernel_driver_active(0):
                self._dev.detach_kernel_driver(0)
        except (usb.core.USBError, NotImplementedError):
            pass  # Not all platforms support this

        # Set configuration and claim interface
        try:
            self._dev.set_configuration()
        except usb.core.USBError:
            pass  # May already be configured

        cfg = self._dev.get_active_configuration()
        intf = cfg[(0, 0)]

        # Find bulk endpoints
        for ep in intf:
            if usb.util.endpoint_direction(ep.bEndpointAddress) == usb.util.ENDPOINT_IN:
                self._ep_in = ep
            elif usb.util.endpoint_direction(ep.bEndpointAddress) == usb.util.ENDPOINT_OUT:
                self._ep_out = ep

        if self._ep_in is None or self._ep_out is None:
            raise ConnectionError("Could not find USB bulk endpoints on Visor")

        logger.info(
            f"Connected to Visor: EP_IN=0x{self._ep_in.bEndpointAddress:02X}, "
            f"EP_OUT=0x{self._ep_out.bEndpointAddress:02X}"
        )

        # Perform CMP handshake
        self._cmp_handshake()

    def _cmp_handshake(self) -> None:
        """Perform CMP handshake — read device init, send our init response."""
        # Read CMP init from device
        raw = self.read(_CMP_SIZE, timeout=10.0)
        if len(raw) < _CMP_SIZE:
            raise ConnectionError(f"CMP init too short: {len(raw)} bytes")

        cmp_type, flags, ver_major, ver_minor, unused, max_baud = struct.unpack(
            _CMP_FORMAT, raw[:_CMP_SIZE]
        )
        logger.info(
            f"CMP init: type={cmp_type}, ver={ver_major}.{ver_minor}, max_baud={max_baud}"
        )

        if cmp_type != CMP_TYPE_INIT:
            raise ConnectionError(f"Expected CMP INIT (1), got type {cmp_type}")

        # Send CMP init response — over USB we don't change baud, just acknowledge
        response = struct.pack(
            _CMP_FORMAT,
            CMP_TYPE_INIT,  # type
            0x00,           # flags (no baud change over USB)
            ver_major,      # match device version
            ver_minor,
            0,              # unused
            0,              # baud rate (ignored over USB)
        )
        self.write(response)
        logger.info("CMP handshake complete")

    def close(self) -> None:
        """Release the USB interface."""
        if self._dev is not None:
            try:
                self._dev.reset()
            except usb.core.USBError:
                pass
            self._dev = None
            self._ep_in = None
            self._ep_out = None

    def read(self, n: int, timeout: float = 5.0) -> bytes:
        """Read up to n bytes from the USB IN endpoint."""
        timeout_ms = int(timeout * 1000)
        try:
            data = self._ep_in.read(n, timeout=timeout_ms)
            return bytes(data)
        except usb.core.USBError as e:
            if "timeout" in str(e).lower():
                raise TimeoutError(f"USB read timeout after {timeout}s") from e
            raise

    def write(self, data: bytes) -> None:
        """Write data to the USB OUT endpoint."""
        self._ep_out.write(data)

    def __enter__(self) -> Connection:
        self.open()
        return self

    def __exit__(self, *args) -> None:
        self.close()
```

"""Desktop Link Protocol (DLP) — high-level commands for PalmOS HotSync."""

from __future__ import annotations

import struct
import logging
from dataclasses import dataclass
from enum import IntEnum

from palm.padp import PADPConnection

logger = logging.getLogger(__name__)


class DLPError(IntEnum):
    """DLP error codes."""
    NONE = 0x0000
    SYSTEM = 0x0001
    MEMORY = 0x0002
    PARAM = 0x0003
    NOT_FOUND = 0x0004
    NONE_OPEN = 0x0005
    ALREADY_OPEN = 0x0006
    TOO_MANY_OPEN = 0x0007
    ALREADY_EXISTS = 0x0008
    OPEN = 0x0009
    DELETED = 0x000A
    BUSY = 0x000B
    NOT_SUPPORTED = 0x000C
    READ_ONLY = 0x000F
    NOT_ENOUGH_SPACE = 0x0010
    LIMIT_EXCEEDED = 0x0011
    CANCELLED = 0x0012


DLP_ERROR_MESSAGES = {
    DLPError.NOT_FOUND: "Database not found",
    DLPError.MEMORY: "Out of memory on device",
    DLPError.ALREADY_EXISTS: "Database already exists",
    DLPError.READ_ONLY: "Database is read-only",
    DLPError.NOT_ENOUGH_SPACE: "Not enough storage space on device",
    DLPError.NOT_SUPPORTED: "Operation not supported",
}


class DLPFuncID(IntEnum):
    """DLP function IDs."""
    READ_SYS_INFO = 0x12
    READ_DB_LIST = 0x16
    OPEN_DB = 0x17
    CREATE_DB = 0x18
    CLOSE_DB = 0x19
    DELETE_DB = 0x1A
    READ_APP_BLOCK = 0x1B
    WRITE_APP_BLOCK = 0x1C
    READ_SORT_BLOCK = 0x1D
    WRITE_SORT_BLOCK = 0x1E
    READ_RECORD = 0x20
    WRITE_RECORD = 0x21
    READ_RESOURCE = 0x22
    WRITE_RESOURCE = 0x23
    READ_OPEN_DB_INFO = 0x28
    OPEN_CONDUIT = 0x2E
    END_OF_SYNC = 0x2F


# DB open modes
DB_MODE_READ = 0x80
DB_MODE_WRITE = 0x40
DB_MODE_READ_WRITE = DB_MODE_READ | DB_MODE_WRITE
DB_MODE_EXCLUSIVE = 0x20

# ReadDBList flags
DBLIST_RAM = 0x80
DBLIST_ROM = 0x40
DBLIST_MULTIPLE = 0x20  # return multiple entries at once


@dataclass
class DLPArg:
    """A DLP request/response argument."""
    arg_id: int
    data: bytes


@dataclass
class SysInfo:
    """Device system info from ReadSysInfo."""
    rom_version: int
    locale: int
    name: str


@dataclass
class DatabaseInfo:
    """Database info from ReadDBList."""
    name: str
    attributes: int
    version: int
    creation_time: int
    modification_time: int
    backup_time: int
    db_type: str
    creator: str
    num_records: int


@dataclass
class Record:
    """A record read from the device."""
    index: int
    attributes: int
    unique_id: int
    data: bytes


@dataclass
class Resource:
    """A resource read from the device."""
    res_type: str
    res_id: int
    index: int
    data: bytes


class DLPException(Exception):
    """Raised when a DLP command returns an error."""
    def __init__(self, func_id: int, error_code: int):
        self.func_id = func_id
        self.error_code = error_code
        msg = DLP_ERROR_MESSAGES.get(error_code, f"Unknown error 0x{error_code:04X}")
        super().__init__(f"DLP error on command 0x{func_id:02X}: {msg}")


def parse_sys_info(data: bytes) -> SysInfo:
    """Parse ReadSysInfo response argument data."""
    rom_version = struct.unpack(">I", data[0:4])[0]
    locale = struct.unpack(">I", data[4:8])[0]
    name = data[8:].split(b"\x00", 1)[0].decode("ascii", errors="replace")
    return SysInfo(rom_version=rom_version, locale=locale, name=name)


class DLPClient:
    """High-level DLP command interface."""

    def __init__(self, padp: PADPConnection):
        self.padp = padp

    @staticmethod
    def build_request(func_id: int, args: list[DLPArg]) -> bytes:
        """Build a DLP request packet."""
        header = struct.pack(">BBB", func_id, len(args), 0x00)
        body = b""
        for arg in args:
            data_len = len(arg.data)
            if data_len <= 0xFF:
                # Small arg: 1-byte id + 1-byte size
                body += struct.pack(">BB", arg.arg_id & 0x3F, data_len)
            else:
                # Long arg: 2-byte id (MSBs=10) + 2-byte size
                body += struct.pack(">HH", (arg.arg_id & 0x3FFF) | 0x8000, data_len)
            body += arg.data
        return header + body

    @staticmethod
    def parse_response(data: bytes) -> tuple[int, int, list[DLPArg]]:
        """Parse a DLP response. Returns (func_id, error_code, args)."""
        if len(data) < 4:
            raise ValueError(f"DLP response too short: {len(data)} bytes")

        func_id = data[0] & 0x7F  # Strip response flag
        arg_count = data[1]
        error_code = struct.unpack(">H", data[2:4])[0]

        args = []
        offset = 4
        for _ in range(arg_count):
            if offset >= len(data):
                break
            first_byte = data[offset]
            if first_byte & 0x80:
                # Long arg: 2-byte id + 2-byte size
                arg_id = struct.unpack(">H", data[offset:offset + 2])[0] & 0x3FFF
                size = struct.unpack(">H", data[offset + 2:offset + 4])[0]
                offset += 4
            else:
                # Small arg: 1-byte id + 1-byte size
                arg_id = first_byte & 0x3F
                size = data[offset + 1]
                offset += 2
            arg_data = data[offset:offset + size]
            args.append(DLPArg(arg_id=arg_id, data=arg_data))
            offset += size

        return func_id, error_code, args

    def _execute(self, func_id: int, args: list[DLPArg] | None = None) -> list[DLPArg]:
        """Send a DLP request and return the response args. Raises on error."""
        args = args or []
        request = self.build_request(func_id, args)
        self.padp.send(request)
        response = self.padp.receive()
        resp_func_id, error_code, resp_args = self.parse_response(response)

        if error_code != 0:
            raise DLPException(func_id, error_code)

        return resp_args

    def read_sys_info(self) -> SysInfo:
        """Read device system info."""
        resp_args = self._execute(DLPFuncID.READ_SYS_INFO)
        if not resp_args:
            raise ValueError("ReadSysInfo returned no data")
        return parse_sys_info(resp_args[0].data)

    def open_conduit(self) -> None:
        """Open conduit — required before database operations."""
        self._execute(DLPFuncID.OPEN_CONDUIT)

    def list_databases(self, ram: bool = True, rom: bool = False) -> list[DatabaseInfo]:
        """List databases on the device."""
        flags = 0
        if ram:
            flags |= DBLIST_RAM
        if rom:
            flags |= DBLIST_ROM
        flags |= DBLIST_MULTIPLE  # request multiple entries per response

        databases = []
        start_index = 0

        while True:
            arg = DLPArg(
                arg_id=0x20,
                data=struct.pack(">BBH", flags, 0, start_index),
            )
            try:
                resp_args = self._execute(DLPFuncID.READ_DB_LIST, [arg])
            except DLPException as e:
                if e.error_code == DLPError.NOT_FOUND:
                    break  # No more databases
                raise

            if not resp_args:
                break

            # Parse response: last_index (2 bytes) + flags (1 byte) + count (1 byte)
            # then count * db_info entries
            resp_data = resp_args[0].data
            last_index = struct.unpack(">H", resp_data[0:2])[0]
            _resp_flags = resp_data[2]
            count = resp_data[3]
            offset = 4

            for _ in range(count):
                # DB info entry: total_size (1 byte) + misc_flags (1 byte) +
                # db_flags (2 bytes) + type (4 bytes) + creator (4 bytes) +
                # version (2 bytes) + modification_num (4 bytes) +
                # crdate (4 bytes) + moddate (4 bytes) + backupdate (4 bytes) +
                # db_index (2 bytes) + name (null-terminated, padded)
                if offset + 32 > len(resp_data):
                    break
                total_size = resp_data[offset]
                db_flags = struct.unpack(">H", resp_data[offset + 2:offset + 4])[0]
                db_type = resp_data[offset + 4:offset + 8].decode("ascii", errors="replace")
                creator = resp_data[offset + 8:offset + 12].decode("ascii", errors="replace")
                version = struct.unpack(">H", resp_data[offset + 12:offset + 14])[0]
                crdate = struct.unpack(">I", resp_data[offset + 18:offset + 22])[0]
                moddate = struct.unpack(">I", resp_data[offset + 22:offset + 26])[0]
                backupdate = struct.unpack(">I", resp_data[offset + 26:offset + 30])[0]
                db_index = struct.unpack(">H", resp_data[offset + 30:offset + 32])[0]
                # Name starts at offset+32 and is null-terminated
                name_start = offset + 32
                name_end = resp_data.index(0, name_start) if 0 in resp_data[name_start:] else len(resp_data)
                name = resp_data[name_start:name_end].decode("ascii", errors="replace")

                databases.append(DatabaseInfo(
                    name=name,
                    attributes=db_flags,
                    version=version,
                    creation_time=crdate,
                    modification_time=moddate,
                    backup_time=backupdate,
                    db_type=db_type,
                    creator=creator,
                    num_records=0,  # not provided in list
                ))

                offset += total_size

            start_index = last_index + 1

        return databases

    def open_db(self, name: str, mode: int = DB_MODE_READ) -> int:
        """Open a database by name. Returns a handle."""
        arg_data = struct.pack(">BB", 0, mode) + name.encode("ascii") + b"\x00"
        resp_args = self._execute(DLPFuncID.OPEN_DB, [DLPArg(arg_id=0x20, data=arg_data)])
        if not resp_args:
            raise ValueError("OpenDB returned no data")
        handle = resp_args[0].data[0]
        return handle

    def close_db(self, handle: int) -> None:
        """Close an open database handle."""
        arg = DLPArg(arg_id=0x20, data=bytes([handle]))
        self._execute(DLPFuncID.CLOSE_DB, [arg])

    def delete_db(self, name: str) -> None:
        """Delete a database by name."""
        arg_data = struct.pack(">B", 0) + name.encode("ascii") + b"\x00"  # card 0 + name
        self._execute(DLPFuncID.DELETE_DB, [DLPArg(arg_id=0x20, data=arg_data)])

    def create_db(self, name: str, creator: str, db_type: str, flags: int, version: int = 0) -> int:
        """Create a new database. Returns a handle."""
        arg_data = struct.pack(
            ">BI4s4sHH",
            0,       # card number
            flags,   # db attributes (unused bytes as creator/type follow)
            creator.encode("ascii")[:4].ljust(4),
            db_type.encode("ascii")[:4].ljust(4),
            version,
            0,       # unused
        ) + name.encode("ascii") + b"\x00"
        resp_args = self._execute(DLPFuncID.CREATE_DB, [DLPArg(arg_id=0x20, data=arg_data)])
        if not resp_args:
            raise ValueError("CreateDB returned no data")
        handle = resp_args[0].data[0]
        return handle

    def read_open_db_info(self, handle: int) -> int:
        """Get the number of records/resources in an open database."""
        resp_args = self._execute(
            DLPFuncID.READ_OPEN_DB_INFO,
            [DLPArg(arg_id=0x20, data=bytes([handle]))],
        )
        if not resp_args:
            raise ValueError("ReadOpenDBInfo returned no data")
        num_records = struct.unpack(">H", resp_args[0].data[:2])[0]
        return num_records

    def read_record(self, handle: int, index: int) -> Record:
        """Read a record by index from an open database."""
        arg_data = struct.pack(">BHH", handle, index, 0)  # handle, index, offset
        arg_data += struct.pack(">H", 0xFFFF)  # max bytes to read (all)
        resp_args = self._execute(DLPFuncID.READ_RECORD, [DLPArg(arg_id=0x20, data=arg_data)])
        if not resp_args:
            raise ValueError("ReadRecord returned no data")
        data = resp_args[0].data
        # Response: record_id (4 bytes) + index (2 bytes) + size (2 bytes) +
        #           attributes (1 byte) + category (1 byte) + record data
        rec_id = struct.unpack(">I", data[0:4])[0]
        rec_index = struct.unpack(">H", data[4:6])[0]
        rec_size = struct.unpack(">H", data[6:8])[0]
        rec_attrs = data[8]
        rec_data = data[10:10 + rec_size]
        return Record(index=rec_index, attributes=rec_attrs, unique_id=rec_id, data=rec_data)

    def write_record(self, handle: int, record: Record) -> None:
        """Write a record to an open database."""
        arg_data = struct.pack(
            ">BIBBH",
            handle,
            record.unique_id,
            record.attributes,
            0,  # category
            len(record.data),
        ) + record.data
        self._execute(DLPFuncID.WRITE_RECORD, [DLPArg(arg_id=0x20, data=arg_data)])

    def read_resource(self, handle: int, index: int) -> Resource:
        """Read a resource by index from an open database."""
        arg_data = struct.pack(">BHH", handle, index, 0)
        arg_data += struct.pack(">H", 0xFFFF)
        resp_args = self._execute(DLPFuncID.READ_RESOURCE, [DLPArg(arg_id=0x20, data=arg_data)])
        if not resp_args:
            raise ValueError("ReadResource returned no data")
        data = resp_args[0].data
        # Response: type (4 bytes) + id (2 bytes) + index (2 bytes) + size (2 bytes) + data
        res_type = data[0:4].decode("ascii", errors="replace")
        res_id = struct.unpack(">H", data[4:6])[0]
        res_index = struct.unpack(">H", data[6:8])[0]
        res_size = struct.unpack(">H", data[8:10])[0]
        res_data = data[10:10 + res_size]
        return Resource(res_type=res_type, res_id=res_id, index=res_index, data=res_data)

    def write_resource(self, handle: int, resource: Resource) -> None:
        """Write a resource to an open database."""
        arg_data = struct.pack(
            ">B4sHH",
            handle,
            resource.res_type.encode("ascii")[:4].ljust(4),
            resource.res_id,
            len(resource.data),
        ) + resource.data
        self._execute(DLPFuncID.WRITE_RESOURCE, [DLPArg(arg_id=0x20, data=arg_data)])

    def read_app_block(self, handle: int) -> bytes:
        """Read the AppInfo block of an open database."""
        arg_data = struct.pack(">BHH", handle, 0, 0xFFFF)  # handle, offset, max_bytes
        resp_args = self._execute(DLPFuncID.READ_APP_BLOCK, [DLPArg(arg_id=0x20, data=arg_data)])
        if not resp_args:
            return b""
        # Response: size (2 bytes) + data
        size = struct.unpack(">H", resp_args[0].data[:2])[0]
        return resp_args[0].data[2:2 + size]

    def write_app_block(self, handle: int, data: bytes) -> None:
        """Write the AppInfo block of an open database."""
        arg_data = struct.pack(">BHH", handle, 0, len(data)) + data
        self._execute(DLPFuncID.WRITE_APP_BLOCK, [DLPArg(arg_id=0x20, data=arg_data)])

    def read_sort_block(self, handle: int) -> bytes:
        """Read the SortInfo block of an open database."""
        arg_data = struct.pack(">BHH", handle, 0, 0xFFFF)
        resp_args = self._execute(DLPFuncID.READ_SORT_BLOCK, [DLPArg(arg_id=0x20, data=arg_data)])
        if not resp_args:
            return b""
        size = struct.unpack(">H", resp_args[0].data[:2])[0]
        return resp_args[0].data[2:2 + size]

    def write_sort_block(self, handle: int, data: bytes) -> None:
        """Write the SortInfo block of an open database."""
        arg_data = struct.pack(">BHH", handle, 0, len(data)) + data
        self._execute(DLPFuncID.WRITE_SORT_BLOCK, [DLPArg(arg_id=0x20, data=arg_data)])

    def end_of_sync(self, status: int = 0) -> None:
        """Signal end of sync session."""
        arg = DLPArg(arg_id=0x20, data=struct.pack(">H", status))
        self._execute(DLPFuncID.END_OF_SYNC, [arg])
```

```python
    @classmethod
    def from_device(cls, dlp, name: str, db_type: str = "", creator: str = "",
                    attributes: int = 0) -> PalmDatabase:
        """Download a database from the device via DLP."""
        from palm.dlp import DB_MODE_READ, DLPException

        is_resource = bool(attributes & ATTR_RESOURCE)
        mode = DB_MODE_READ
        handle = dlp.open_db(name, mode)

        try:
            num_items = dlp.read_open_db_info(handle)

            db = cls(
                name=name,
                attributes=attributes,
                db_type=db_type,
                creator=creator,
            )

            # Read app info
            app_info = dlp.read_app_block(handle)
            if app_info:
                db.app_info = app_info

            # Read sort info
            sort_info = dlp.read_sort_block(handle)
            if sort_info:
                db.sort_info = sort_info

            # Read records or resources
            for i in range(num_items):
                if is_resource:
                    res = dlp.read_resource(handle, i)
                    db.resources.append(Resource(
                        res_type=res.res_type, res_id=res.res_id, data=res.data,
                    ))
                else:
                    rec = dlp.read_record(handle, i)
                    db.records.append(Record(
                        data=rec.data, attributes=rec.attributes, unique_id=rec.unique_id,
                    ))

            return db
        finally:
            dlp.close_db(handle)

    def to_device(self, dlp) -> None:
        """Upload this database to the device via DLP."""
        from palm.dlp import DLPException, DLPError, DB_MODE_READ_WRITE

        # Delete existing DB (ignore if not found)
        try:
            dlp.delete_db(self.name)
        except DLPException as e:
            if e.error_code != DLPError.NOT_FOUND:
                raise

        # Create the database
        handle = dlp.create_db(
            name=self.name,
            creator=self.creator,
            db_type=self.db_type,
            flags=self.attributes,
            version=self.version,
        )

        try:
            # Write app info
            if self.app_info:
                dlp.write_app_block(handle, self.app_info)

            # Write sort info
            if self.sort_info:
                dlp.write_sort_block(handle, self.sort_info)

            # Write records or resources
            if self.is_resource_db:
                for res in self.resources:
                    from palm.dlp import Resource as DLPResource
                    dlp.write_resource(handle, DLPResource(
                        res_type=res.res_type, res_id=res.res_id, index=0, data=res.data,
                    ))
            else:
                for rec in self.records:
                    from palm.dlp import Record as DLPRecord
                    dlp.write_record(handle, DLPRecord(
                        index=0, attributes=rec.attributes,
                        unique_id=rec.unique_id, data=rec.data,
                    ))
        finally:
            dlp.close_db(handle)
```

"""CLI entry point for palm-com — PalmOS device communication tool."""

from __future__ import annotations

import os
import sys
import logging

import click

from palm.transport import Connection
from palm.slp import SLPSocket
from palm.padp import PADPConnection
from palm.dlp import DLPClient, DatabaseInfo, DB_MODE_READ
from palm.pdb import PalmDatabase, ATTR_RESOURCE


def setup_logging(verbose: bool) -> None:
    level = logging.DEBUG if verbose else logging.WARNING
    logging.basicConfig(
        level=level,
        format="%(name)s: %(message)s",
    )


class DeviceSession:
    """Context manager that opens a full HotSync session."""

    def __init__(self, port: str | None = None, verbose: bool = False):
        self.port = port
        self.verbose = verbose
        self.conn: Connection | None = None
        self.dlp: DLPClient | None = None

    def __enter__(self) -> DLPClient:
        self.conn = Connection()
        self.conn.open()

        slp = SLPSocket(self.conn)
        padp = PADPConnection(slp)
        self.dlp = DLPClient(padp)

        # Required handshake before DB operations
        self.dlp.open_conduit()

        return self.dlp

    def __exit__(self, *args) -> None:
        if self.dlp is not None:
            try:
                self.dlp.end_of_sync()
            except Exception:
                pass
        if self.conn is not None:
            self.conn.close()


@click.group()
@click.option("--verbose", "-v", is_flag=True, help="Show protocol debug output")
@click.option("--port", default=None, help="Override USB device (unused for USB)")
@click.pass_context
def cli(ctx, verbose, port):
    """palm-com — communicate with PalmOS devices over USB."""
    ctx.ensure_object(dict)
    ctx.obj["verbose"] = verbose
    ctx.obj["port"] = port
    setup_logging(verbose)


@cli.command()
@click.pass_context
def sysinfo(ctx):
    """Show device system info."""
    with DeviceSession(verbose=ctx.obj["verbose"]) as dlp:
        info = dlp.read_sys_info()
        rom_major = (info.rom_version >> 24) & 0xFF
        rom_minor = (info.rom_version >> 20) & 0x0F
        click.echo(f"Device: {info.name}")
        click.echo(f"ROM Version: {rom_major}.{rom_minor}")


@cli.command("list")
@click.option("--rom", is_flag=True, help="Include ROM databases")
@click.option("--ram/--no-ram", default=True, help="Include RAM databases")
@click.pass_context
def list_dbs(ctx, rom, ram):
    """List databases on device."""
    with DeviceSession(verbose=ctx.obj["verbose"]) as dlp:
        databases = dlp.list_databases(ram=ram, rom=rom)
        if not databases:
            click.echo("No databases found.")
            return

        # Print table header
        click.echo(f"{'Name':<32} {'Type':<6} {'Creator':<8} {'Flags':<6}")
        click.echo("-" * 54)
        for db in databases:
            flag_str = "R" if db.attributes & ATTR_RESOURCE else "D"
            click.echo(f"{db.name:<32} {db.db_type:<6} {db.creator:<8} {flag_str:<6}")
        click.echo(f"\n{len(databases)} database(s) found.")


@cli.command()
@click.argument("name")
@click.pass_context
def info(ctx, name):
    """Show database header info."""
    with DeviceSession(verbose=ctx.obj["verbose"]) as dlp:
        databases = dlp.list_databases(ram=True, rom=True)
        db_info = next((d for d in databases if d.name == name), None)
        if db_info is None:
            click.echo(f"Database '{name}' not found.", err=True)
            sys.exit(1)

        click.echo(f"Name:    {db_info.name}")
        click.echo(f"Type:    {db_info.db_type}")
        click.echo(f"Creator: {db_info.creator}")
        click.echo(f"Version: {db_info.version}")
        is_res = "Yes (.prc)" if db_info.attributes & ATTR_RESOURCE else "No (.pdb)"
        click.echo(f"Resource DB: {is_res}")

        handle = dlp.open_db(name, DB_MODE_READ)
        try:
            num = dlp.read_open_db_info(handle)
            label = "Resources" if db_info.attributes & ATTR_RESOURCE else "Records"
            click.echo(f"{label}: {num}")
        finally:
            dlp.close_db(handle)


@cli.command()
@click.argument("name")
@click.option("--out", default=None, help="Output file path")
@click.pass_context
def pull(ctx, name, out):
    """Download a database from the device."""
    with DeviceSession(verbose=ctx.obj["verbose"]) as dlp:
        # Find database info
        databases = dlp.list_databases(ram=True, rom=True)
        db_info = next((d for d in databases if d.name == name), None)
        if db_info is None:
            click.echo(f"Database '{name}' not found.", err=True)
            sys.exit(1)

        click.echo(f"Downloading '{name}'...")
        db = PalmDatabase.from_device(
            dlp, name=name,
            db_type=db_info.db_type,
            creator=db_info.creator,
            attributes=db_info.attributes,
        )

        if out is None:
            ext = ".prc" if db.is_resource_db else ".pdb"
            out = name + ext

        db.to_file(out)
        items = len(db.resources) if db.is_resource_db else len(db.records)
        label = "resources" if db.is_resource_db else "records"
        click.echo(f"Saved {items} {label} to {out}")

    click.echo(f"Done — '{db.name}' installed on device.")

        click.echo(f"Deleted '{name}'.")
